/**
 * MongoDB Collections Setup Script
 * 
 * DATABASE: recruitmentDB
 * Collections:
 * - runs             (embeds runConfig snapshot per run)
 * - jobs
 * - companies
 * - prospects
 * - outreach
 * - icpConfig        (single versioned doc: titles, locations, industries, personaMappings)
 * - outreachTemplates
 * 
 * NOTE: buyerPersonaMappings is DEPRECATED — folded into icpConfig.personaMappings
 * NOTE: Title/industry filter keywords stay in config.py (not stored in DB)
 * 
 * NAMING CONVENTIONS APPLIED:
 * ✓ Collections: lowercase, plural nouns (runs, jobs, companies, etc.)
 * ✓ Fields: camelCase (firstName, createdAt, companyId)
 * ✓ Foreign Keys: camelCase with 'Id' suffix (jobId, companyId, prospectId)
 * ✓ Indexes: idx_fieldName format for clarity
 * ✓ All field names avoid special characters and dots
 * ✓ Maximum namespace length maintained (< 120 bytes)
 */

// ============================================================================
// 1. RUNS COLLECTION
// ============================================================================
// Purpose: Track recruitment run campaigns/batches
// runConfig embeds the exact ICP snapshot used — so historical runs are
// self-contained even if icpConfig changes in the future.

db.createCollection("runs", {
  validator: {
    $jsonSchema: {
      bsonType: "object",
      title: "Runs",
      required: ["_id", "title", "runStartedAt", "runConfig"],
      properties: {
        _id: {
          bsonType: "objectId",
          description: "Unique identifier for the run"
        },
        title: {
          bsonType: "string",
          minLength: 1,
          maxLength: 255,
          description: "Human-readable label for this run"
        },
        source: {
          enum: ["jobspy", "linkedin_api", "mixed"],
          description: "Data source that initiated this run"
        },
        runStartedAt: {
          bsonType: "date",
          description: "Start date/time of the run"
        },
        runEndedAt: {
          bsonType: "date",
          description: "End date/time of the run (set on completion)"
        },
        status: {
          enum: ["active", "completed", "paused", "cancelled"],
          description: "Current status of the run"
        },
        stats: {
          bsonType: "object",
          description: "Live counters updated as data flows in during the run",
          properties: {
            totalJobsScraped:  { bsonType: "int", description: "Total raw jobs scraped" },
            inserted:          { bsonType: "int", description: "New jobs inserted (not duplicates)" },
            duplicates:        { bsonType: "int", description: "Duplicate jobs skipped" },
            acceptedJobs:      { bsonType: "int", description: "Jobs that passed title rejection" },
            rejectedJobs:      { bsonType: "int", description: "Jobs rejected by title filter" },
            uniqueCompanies:   { bsonType: "int", description: "Unique companies found" },
            acceptedCompanies: { bsonType: "int", description: "Companies that passed rejection engine" },
            rejectedCompanies: { bsonType: "int", description: "Companies filtered out" },
            skippedCompanies:  { bsonType: "int", description: "Companies skipped (no LinkedIn data)" },
            totalProspects:    { bsonType: "int", description: "Total qualified prospects found" }
          }
        },
        runConfig: {
          bsonType: "object",
          required: ["searchTitles", "searchLocations"],
          description: "Snapshot of the ICP config used for this run",
          properties: {
            searchTitles: {
              bsonType: "array",
              items: { bsonType: "string" },
              description: "Job titles searched in this run"
            },
            searchLocations: {
              bsonType: "array",
              items: { bsonType: "string" },
              description: "Locations searched in this run"
            },
            hoursOld: {
              bsonType: "int",
              description: "Max age of job postings to include"
            },
            resultsPerSearch: {
              bsonType: "int",
              description: "Max results fetched per title x location combo"
            },
            siteName: {
              bsonType: "array",
              items: { bsonType: "string" },
              description: "Job boards used e.g. ['linkedin']"
            },
            icpConfigSnapshot: {
              bsonType: ["object", "null"],
              description: "Snapshot of icpConfig _id and version used",
              properties: {
                icpConfigId: { bsonType: ["objectId", "null"] },
                version:     { bsonType: "int" }
              }
            }
          }
        },
        createdAt: {
          bsonType: "date",
          description: "When this document was created"
        },
        updatedAt: {
          bsonType: "date",
          description: "When this document was last updated"
        }
      }
    }
  }
});

// Create indexes for runs collection
db.runs.createIndex({ "title": 1 }, { name: "idx_title" });
db.runs.createIndex({ "source": 1 }, { name: "idx_source" });
db.runs.createIndex({ "runStartedAt": 1 }, { name: "idx_runStartedAt" });
db.runs.createIndex({ "status": 1 }, { name: "idx_status" });
db.runs.createIndex({ "runConfig.icpConfigSnapshot.version": 1 }, { name: "idx_icpVersion" });
db.runs.createIndex({ "createdAt": -1 }, { name: "idx_createdAt_desc" });

// SUMMARY OF BEST PRACTICES IMPLEMENTED

/*
 * ✓ NAMING CONVENTIONS:
 *   - Collections: Lowercase, plural (runs, jobs, companies, prospects, outreach, etc.)
 *   - Fields: camelCase (firstName, companyId, createdAt)
 *   - Foreign Keys: Suffixed with 'Id' (jobId, companyId, prospectId)
 *   - Indexes: Prefixed with 'idx_' for clarity
 *   - Timestamps: createdAt and updatedAt for audit trails
 *
 * ✓ SCHEMA VALIDATION:
 *   - JSON Schema validators for all collections
 *   - Required fields explicitly defined
 *   - Data types enforced (bsonType)
 *   - String length constraints to optimize storage
 *   - Email and domain pattern validation
 *   - Enum values for status fields
 *
 * ✓ INDEXING STRATEGY:
 *   - Single field indexes on frequently queried fields
 *   - Composite indexes for common query patterns (e.g., runId + createdAt)
 *   - Unique indexes where appropriate (email, externalId, jobId+prospectId)
 *   - Descending indexes on date fields for latest-first queries
 *   - Indexes on foreign key fields for JOIN-like operations
 *
 * ✓ RELATIONSHIPS:
 *   - Foreign key fields clearly named with 'Id' suffix
 *   - Comments document the relationships
 *   - No embedded arrays of references (maintains flexibility)
 *   - runs.runConfig embeds ICP snapshot for historical integrity
 *
 * ✓ NESTED JSON:
 *   - Used for extensible data (jobDetails, prospectDetails, companyDetails)
 *   - runConfig embedded in runs for self-contained run history
 *   - icpConfig uses arrays of sub-documents for titles, locations, industries
 *   - Allows flexibility without schema changes
 *   - Still validated at top level
 *
 * ✓ PERFORMANCE:
 *   - Shorter field names reduce document size
 *   - Strategic indexes prevent full collection scans
 *   - Descending date indexes support pagination
 *   - Composite indexes for multi-criteria queries
 *
 * ✓ MAINTAINABILITY:
 *   - Clear collection purposes documented
 *   - Field descriptions in schema
 *   - Explicit index naming reveals intent
 *   - Consistent timestamp naming (createdAt, updatedAt)
 *   - Filter keywords (accept/reject) kept in config.py — not duplicated in DB
 *   - buyerPersonaMappings deprecated and folded into icpConfig.personaMappings
 */
